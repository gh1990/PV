' Copiază aceste proceduri în modulul ThisWorkbook al workbook-ului PV.xlsm

Private Sub Workbook_Open()
    On Error Resume Next
    Wrappers.EnsureAddinLoaded
    Application.Run "'PV_Addin.xlam'!InitHost", Me
    On Error GoTo 0
End Sub

Private Sub Workbook_SheetChange(ByVal Sh As Object, ByVal Target As Range)
    On Error Resume Next
    Application.Run "'PV_Addin.xlam'!Handle_SheetChange", Sh, Target
    On Error GoTo 0
End Sub

Private Sub Workbook_SheetActivate(ByVal Sh As Object)
    On Error Resume Next
    Application.Run "'PV_Addin.xlam'!Handle_SheetActivate", Sh
    On Error GoTo 0
End Sub
